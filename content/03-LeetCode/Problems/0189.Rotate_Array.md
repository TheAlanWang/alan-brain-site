```python
class Solution:
    def rotate(self, nums: List[int], k: int) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """        
        n = len(nums)
        k = k % n
        if k == 0:
            return
            
        rotate_part = nums[-k:] 
        remain = nums[:-k]
        
        nums[:k] = rotate_part 
        nums[k:] = remain
        return nums
```
