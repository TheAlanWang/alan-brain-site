```python
class Solution:
    def canJump(self, nums: List[int]) -> bool:
        n = len(nums)

        farthest = 0
        for i, step in enumerate(nums):
            if i > farthest:
                return False
            
            else:
                farthest = max(farthest, i + step)
                if farthest >= n - 1:
                    return True
```
